## 获取指标数据

<GET>/cron/dashboard/stats</GET>

### 请求

<TypeTable
  type={{
    taskType: {
      description: '任务类型过滤',
      type: "string<'all' | 'user' | 'system'>",
      default: 'all',
    },
  }}
/>

### 响应

<TypeTable
  noRequiredMark={true}
  type={{
    todaySuccessCount: {
      description: '今日成功执行的任务数量',
      type: 'number',
      required: true,
    },
    todayFailureCount: {
      description: '今日执行失败的任务数量',
      type: 'number',
      required: true,
    },
    yesterdaySuccessCount: {
      description: '昨日成功执行的任务数量',
      type: 'number',
      required: true,
    },
    yesterdayFailureCount: {
      description: '昨日执行失败的任务数量',
      type: 'number',
      required: true,
    },
    enabledCount: {
      description: '当前启用的任务数量',
      type: 'number',
      required: true,
    },
    disabledCount: {
      description: '当前禁用的任务数量',
      type: 'number',
      required: true,
    },
    runningCount: {
      description: '当前正在运行的任务数量',
      type: 'number',
      required: true,
    },
  }}
/>

```json title="示例"
{
    "todaySuccessCount": 8,
    "todayFailureCount": 0,
    "yesterdaySuccessCount": 13,
    "yesterdayFailureCount": 1,
    "enabledCount": 18,
    "disabledCount": 23,
    "runningCount": 0
}
```

## 获取任务运行趋势

<GET>/cron/dashboard/trend</GET>

### 请求

<TypeTable
  type={{
    taskType: {
      description: '任务类型过滤',
      type: "string<'all' | 'user' | 'system'>",
      default: 'all',
    },
    date: {
      description: '日期过滤，默认不填为今天',
      type: "string<'yyyy-MM-dd'>",
      default: '今天',
    },
  }}
/>

### 响应

返回数组对象

<TypeTable
  noRequiredMark={true}
  type={{
    timestamp: {
      description: '执行时间戳',
      type: 'number',
      required: true,
    },
    taskId: {
      description: '任务ID',
      type: 'number',
      required: true,
    },
    taskName: {
      description: '任务名称',
      type: 'string',
      required: true,
    },
    taskType: {
      description: '任务类型',
      type: "string<'system' | 'user'>",
      required: true,
    },
    duration: {
      description: '运行时长，单位毫秒',
      type: 'number',
      required: true,
    },
    success: {
      description: '运行结果（是否成功）',
      type: 'boolean',
      required: true,
    },
  }}
/>

```json title="示例"
[
    {
        "timestamp": 1773161661000,
        "taskId": 1,
        "taskName": "更新仓库",
        "taskType": "user",
        "duration": 21749,
        "success": true
    },
    {
        "timestamp": 1773233647000,
        "taskId": 1,
        "taskName": "更新仓库",
        "taskType": "user",
        "duration": 7042,
        "success": true
    }
    ...
]
```

## 获取正在运行中的任务

<GET>/cron/dashboard/running</GET>

### 请求

<TypeTable
  type={{
    taskType: {
      description: '任务类型过滤',
      type: "string<'all' | 'user' | 'system'>",
      default: 'all',
    },
  }}
/>

### 响应

返回数组对象

<TypeTable
  noRequiredMark={true}
  type={{
    id: {
      description: '任务ID',
      type: 'number',
      required: true,
    },
    name: {
      description: '任务名称',
      type: 'string',
      required: true,
    },
    type: {
      description: '任务类型',
      type: "string<'system' | 'user'>",
      required: true,
    },
    cron: {
      description: '任务的cron表达式',
      type: 'string',
      required: true,
    },
    shell: {
      description: '任务的执行命令',
      type: 'string',
      required: true,
    },
  }}
/>

```json title="示例"
[
    {
        "id": 1,
        "name": "更新仓库",
        "type": "user",
        "cron": "54 */2 * * *",
        "shell": "arcadia update repo"
    },
    {
        "id": 2,
        "name": "清理日志",
        "type": "system",
        "cron": "48 5 * * *",
        "shell": "arcadia rmlog 7"
    }
    ...
]
