import { POST } from '@/components/RequestTag';

<POST>/user/auth</POST>

## 请求

<TypeTable
  type={{
    username: {
      description: '用户名',
      type: 'string',
      required: true,
    },
    password: {
      description: '密码',
      type: 'string',
      required: true,
    },
    captcha: {
      description: '图形验证码',
      type: 'string',
    },
  }}
/>

## 响应

<TypeTable
  noRequiredMark={true}
  type={{
    token: {
      description: 'JWT Token',
      type: 'string',
      required: true,
    },
    newPwd: {
      description: '系统自动更改的新密码，返回机制见下方注释',
      type: 'string',
      required: true,
    },
    showCaptcha: {
      description: '是否需要提供图形验证码',
      type: 'boolean',
      required: true,
    },
    limitTime: {
      description: '限制登录剩余时长（单位/秒）',
      type: 'number',
      required: true,
    },
    requireTwoFactor: {
      description: '是否启用双重认证',
      type: 'boolean',
      required: true,
    },
  }}
/>

```json title="示例"
{
  "token": "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789",
  "newPwd": ""
}
```

Token 有效时长：72小时

<Callout title="注释">
  `newPwd` 仅在系统强制更改用户密码后才返回，根据策略目前仅判断用户密码是否为初始密码，如果是则返回一个由随机字符组成的新密码，否则返回空字符串，密码变更后管理面板会有通知提醒。  
  基础参数验证通过后就会响应 `success` 成功，如果启用了双重认证则不会返回 `token`，需要继续调用下方的接口。
</Callout>
