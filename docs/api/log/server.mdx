## 请求

<TypeTable
  type={{
    page: {
      description: '指定页数',
      type: 'string',
      default: 1,
    },
    size: {
      description: '指定分页大小数量',
      type: 'string',
      default: 20,
    },
    search: {
      description: '日志内容过滤',
      type: 'string',
    },
    type: {
      description: '日志类型过滤',
      type: "string<'trace' | 'debug' | 'info' | 'warn' | 'error' | 'fatal'>",
      typeDescription: '多个类型用英文逗号分隔',
    },
    order: {
      description: '指定时间列的排序方式',
      typeDescription: '0 升序，1 降序',
      type: "string<'0' | '1'>",
    },
  }}
/>
